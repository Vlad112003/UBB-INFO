#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <string.h>

int main(int argc, char** argv){
        int nr;
        char simbol;
        //deschid fisier pt citit
        if(argc!=2)
        {
                perror("Dati un fisier ca argument!!");
                exit(EXIT_FAILURE);
        }
        FILE* fd;
        fd=fopen(argv[1],"r");
        if(fd==NULL)
        {
                perror("fopen()");
                exit(EXIT_FAILURE);
        }

        //fac pipe
        int f[2];
        int res=pipe(f);
        if(res==-1)
        {
                perror("pipe()");
                exit(EXIT_FAILURE);
        }
        int pid=fork();
        if(pid==-1)
        {
                perror("fork()");
                exit(EXIT_FAILURE);
        }
        //IN COPIL
        if(pid==0)
        {
                while(1)
                {       int sum=0,prod=1,rez;
                        read(f[0],&simbol,sizeof(char));
                        if(simbol=='0') break;
                        read(f[0],&nr,sizeof(int));
                        int copy=nr;
                        if(simbol=='+')
                        {
                                while(nr)
                                {
                                        sum+=nr%10;
                                        nr=nr/10;
                                }
                                rez=sum;
                        }
                        if(simbol=='*')
                        {
				 while(nr)
                                {
                                        prod*=nr%10;
                                        nr=nr/10;
                                }
                                rez=prod;
                        }
                        printf("%c %d => %d\n",simbol,copy,rez);

                }

                close(f[1]);
                close(f[0]);
                exit(0);
        }


        //IN PARINTE
        while(fscanf(fd,"%c %d\n",&simbol,&nr)!=-1)
        {
                write(f[1],&simbol,sizeof(char));
                write(f[1],&nr,sizeof(int));
        }
        simbol='0';
        write(f[1],&simbol,sizeof(char));
        close(f[0]);
        close(f[1]);
        fclose(fd);
	int status;
	wait(&status);
	return 0;
}
