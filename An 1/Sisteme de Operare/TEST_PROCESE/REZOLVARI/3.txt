#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <string.h>

int main()
{       char ch,s[20];
        int cnt=0,i;
        int f[2];
        int res=pipe(f);
        if (res==-1)
        {
                perror("pipe()");
                exit(EXIT_FAILURE);
        }

        int a;
        int pid=fork();
        if(pid==0)
        {
                close(f[1]);
                read(f[0],&ch,sizeof(char));
                read(f[0],&a,sizeof(int));
                read(f[0],&s,a*sizeof(char));

                for(i=0;i<a;i++)
                {
                        if(s[i]==ch)
                                cnt++;
                }

                close(f[0]);
                printf("[CHILD] %d\n",cnt);
                exit(cnt);

        }

        //acum sunt in parinte
        close(f[0]);
        scanf("%c",&ch);
        write(f[1],&ch,sizeof(char));
        scanf("%s",s);
        a=strlen(s);
        write(f[1],&a,sizeof(int));
        write(f[1],&s,strlen(s)*sizeof(char));

        int status;
        wait(&status); //astept sa termine copilul
        printf("[PARENT] Copilul a terminat cu exit status %d\n",status);
        close(f[1]);

        return 0;
}
