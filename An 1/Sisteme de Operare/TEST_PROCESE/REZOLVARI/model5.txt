#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <string.h>

int main(int argc, char** argv){
        char ch,sir[100];
        int i;
        //deschid fisier pt citit
        if(argc!=2)
        {
                perror("Dati un fisier ca argument!!");
                exit(EXIT_FAILURE);
        }
        FILE* fd;
        fd=fopen(argv[1],"r");
        if(fd==NULL)
        {
                perror("fopen()");
                exit(EXIT_FAILURE);
        }

        //fac pipe
        int f[2];
        int res=pipe(f);
        if(res==-1)
        {
                perror("pipe()");
                exit(EXIT_FAILURE);
        }
        int pid=fork();
        if(pid==-1)
        {
                perror("fork()");
                exit(EXIT_FAILURE);
        }
        //IN COPIL
        if(pid==0)
        {       int poz=0;
                while(1)
                {
                        read(f[0],&ch,sizeof(char));
                        if(ch=='0') break;
                        read(f[0],&sir,sizeof(sir));

                        for(i=0;i<strlen(sir);i++)
                                if(sir[i]==ch) poz=i+1;
                        if(poz==0) poz=-1;
                        printf("%c %s pozitia:%d\n",ch,sir,poz);

                }

                close(f[1]);
                close(f[0]);
                exit(0);
        }
	 //IN PARINTE
        while(fscanf(fd,"%c %s\n",&ch,sir)!=-1)
        {
                write(f[1],&ch,sizeof(char));
                write(f[1],sir,sizeof(sir));
        }
        ch='0';
        write(f[1],&ch,sizeof(char));
        close(f[0]);
        close(f[1]);
        fclose(fd);
        int status;
        wait(&status);
        return 0;
}
