
// // writer.c // // Using named pipe (FIFO) - Create and write to a FIFO //
#include <fcntl.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <unistd.h>
int main(int argc, char* argv[]) {
        int n=0;
    char* fifo1 = "./fifo1";
    // create the FIFO (named pipe)
    int res = mkfifo(fifo1, 0644);
    if (res == -1)
    {
        perror("mkfifo()");
        exit(EXIT_FAILURE);
    }
    // open the FIFO for write
    int fd = open(fifo1, O_WRONLY);
    if (fd == -1)
    {
        perror("open()");
        exit(EXIT_FAILURE);
    }


        FILE* fp;
        fp=fopen(argv[1],"r");
        char ch;
        while(fscanf(fp,"%c",&ch)!=-1)
        {if(ch<='9' && ch>='0') //e cifra
                {       int nr;
                        nr=ch-'0';
                        n=n*10+nr;
                }
        }
         write(fd,&n,sizeof(int));
        fclose(fp);
    close(fd);
    // remove the FIFO
    unlink(fifo1);
    return 0;
}
