#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
int prim(int a){
int i;
for(i=2;i<=a/2;i++)
{
if(a%i==0) return 0;
}
return 1;

}

int main() {

int f[2];
int res=pipe(f);

if (res==-1)
{
perror("pipe()");
exit(EXIT_FAILURE);
}

int pid=fork();
if(pid==-1)
{
perror("fork()");
exit(EXIT_FAILURE);
}
int a;
if(pid==0)      //in copil
{
close(f[1]);
read(f[0],&a,sizeof(int));

if(prim(a)==1) printf("Numarul %d este prim!",a);
else printf("Numarul %d nu este prim!",a);

close(f[0]);
exit(0);
}

close(f[0]);

scanf("%d",&a);
write(f[1],&a,sizeof(int));

int status;
wait(&status);
close(f[1]);

return 0;
}
